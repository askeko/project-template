\newcounter{Rev} % Revision counter
\setcounter{Rev}{13} % Set the revision counter to the number of the current revision

% Add text to the document: \add{Revision number}{text} - Text will be blue in the revision and black in other revisions.
\newcommand{\add}[2]{\protect\ifthenelse{#1=\value{Rev}}{\textcolor{blue}{#2}}{#2}}

% Remove text to the document: \rmv{Revision number}{text} - Text will be red and strikethrough in the revision and removed from other revisions.
\newcommand{\rmv}[2]{\protect\ifthenelse{#1=\value{Rev}}{\textcolor{red}{\sout{#2}}}{}}

% For changing text: \change{Revision number}{text1}{text2} - Equal to \rmv{Revision number}{text1} \add{Revision number}{text2}.
\newcommand{\change}[3]{\protect\ifthenelse{#1=\value{Rev}}{\textcolor{red}{\sout{#2}} \textcolor{blue}{#3}}{#3}}

% For modifying code blocks like figures or tables where the \add and \rem commands do not work: \changecode{Revision number}{text1}{text2} - Text1 will be used in the revision and text2 is used in other revisions.
\newcommand{\changecode}[3]{\protect\ifthenelse{#1=\value{Rev}}{#2}{#3}}
